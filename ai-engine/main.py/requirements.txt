from fastapi import FastAPI, HTTPException, Depends
from pydantic import BaseModel
import litellm # The ultimate router for OpenAI, DeepSeek, Gemini, Meta
import os
from dotenv import load_dotenv

# AWS Server के लिए Environment Load करें
load_dotenv()

app = FastAPI(title="TriNetra Super AI Brain")

# ─── 1. REAL API KEYS VALIDATION ───
# AWS पर ये Keys मौजूद होनी चाहिए
os.environ["OPENAI_API_KEY"] = os.getenv("OPENAI_API_KEY", "")
os.environ["GEMINI_API_KEY"] = os.getenv("GEMINI_API_KEY", "")
os.environ["GROQ_API_KEY"] = os.getenv("GROQ_API_KEY", "") # For Meta Llama fast inference
os.environ["DEEPSEEK_API_KEY"] = os.getenv("DEEPSEEK_API_KEY", "")

# ─── 2. DATA SECURITY SCHEMA ───
class AIPromptRequest(BaseModel):
    prompt: str
    mode: str # "A", "B", "C", "OS"
    user_id: str

# ─── 3. THE 6-IN-1 SUPER BRAIN CONTROLLER ───
@app.post("/api/v1/brain/process")
async def process_super_ai(request: AIPromptRequest):
    try:
        model_to_use = ""
        
        # 🧠 MODE A: Basic AI (Cheaper, Fast - Meta Llama via Groq)
        if request.mode == "A":
            model_to_use = "groq/llama3-70b-8192" 
            
        # 🧠 MODE B: Advanced AI (Gemini Pro - Good for reasoning)
        elif request.mode == "B":
            model_to_use = "gemini/gemini-1.5-pro"
            
        # 🧠 MODE C: Master AI (The Ultimate 6-in-1 Brain - GPT-4o or DeepSeek)
        elif request.mode == "C":
            # You can dynamically route this based on who answers better/cheaper
            model_to_use = "gpt-4o" 
            
        else:
            raise HTTPException(status_code=400, detail="Invalid TriNetra AI Mode")

        # 🚨 REAL AI EXECUTION VIA LITELLM (No Dummy Code)
        response = litellm.completion(
            model=model_to_use,
            messages=[
                {"role": "system", "content": "You are TriNetra Master AI, an advanced intelligence engine created by Suryavanshi Kalki."},
                {"role": "user", "content": request.prompt}
            ],
            temperature=0.7
        )

        # Return the pure output to the Node.js Server
        return {
            "status": "success",
            "ai_response": response.choices[0].message.content,
            "tokens_used": response.usage.total_tokens,
            "engine_used": model_to_use
        }

    except Exception as e:
        print(f"❌ [AI Brain Crash]: {str(e)}")
        raise HTTPException(status_code=500, detail="TriNetra AI Engine Overload or Error")

# Start Command for AWS: uvicorn main:app --host 0.0.0.0 --port 8000
